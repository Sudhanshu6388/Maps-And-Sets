Ques.1 You are given a string (STR) of length N, consisting of only the lower case English alphabet. Your task is to remove all the duplicate occurrences of characters in the string.

Input: abcadeecfb

Output: abcdef

            let myStr ='abcadeecfb';
            myStrArr= myStr.split('');
            
            for(let i = 0 ; i < myStrArr.length ; i++){
                for(let j =i+1 ; j < myStrArr.length ; j++){
                    if(myStrArr[i] === myStrArr[j]){
                      myStrArr.splice(j, 1);
                    }
                }    
            }
            myStr = myStrArr.join("");
            
            console.log('After Deletion : ',myStr);
